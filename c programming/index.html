int n,rem,rev = 0;
printf("enter any integer: \n");
scanf("%d",&n);
while(n!=0){
rem = n % 10;
rev = rev * 10 + rem;
n = n/10;
}
if(n == rev){
printf("palindrome");

if(x == 0){
return 0;
}else if(x == 1){
return 1;
}else{
return(x*fact(x-1));
}
